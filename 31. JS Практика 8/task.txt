//(2, 9) - сумма чисел от 2 до 9
//(32, 88) - сумма чисел от 32 до 88

function getSum(min, max) {
   let sum = 0;
   for (let i = min; i <= max; i++) {
     sum += i;
   }
   return sum;
}
